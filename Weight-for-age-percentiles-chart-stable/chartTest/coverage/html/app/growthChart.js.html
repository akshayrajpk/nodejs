         function growthCheck(index){
            if(parseFloat(plotPoint.weight) <= parseFloat(data[index].P3)){
              summary = "Growth rate 3% or below";
            }
            else if(parseFloat(plotPoint.weight) > parseFloat(data[index].P3)  && parseFloat(plotPoint.weight) <= parseFloat(data[index].P50)){
              summary = "Growth rate between 3% & 50%";
            }
            else if(parseFloat(plotPoint.weight) > parseFloat(data[index].P50)  && parseFloat(plotPoint.weight) <= parseFloat(data[index].P97)){
                    summary = "Growth rate between 50% & 97%";
            }
            else Eif(parseFloat(plotPoint.weight) > parseFloat(data[index].P97)){
              summary = "Growth rate above 97%";
            }  
          }//end of growthcheck
 
        function growthCheckMidValues(index){
          function average(first, second){
            var sum = parseFloat(first)+parseFloat(second);
            return sum/2;
          }      
          if(parseFloat(plotPoint.weight) <= average(data[index].P3,data[index+1].P3)){
            summary = "Growth rate 3% or below";
          }
          else if(parseFloat(plotPoint.weight) > average(data[index].P3, data[index+1].P3)  && parseFloat(plotPoint.weight) <= average(data[index].P50 , data[index+1].P50)){
            summary = "Growth rate between 3% & 50%";
          }
          else if(parseFloat(plotPoint.weight) > average(data[index].P50,data[index+1].P50)  && parseFloat(plotPoint.weight) <= average(data[index].P97,data[index+1].P97)){
            summary = "Growth rate between 50% & 97%";
          }
          else Eif(parseFloat(plotPoint.weight) > average(data[index].P97,data[index+1].P97)){
            summary = "Growth rate above 97%";
          }  
        }//end of growthcheckMidValues
