function drawLine(svg, data, color, Line_width)
{
 
  //console.log("Line");
  var line =d3.line()
   .x(function(d) {return (((d.month *1)*((width-margin.right)/39))+50.5)}) // set the x values for the line generator
   .y(function(d) {return (height-margin.top) - (d.value * ((height-margin.top)/42))+60.5}) // set the y values for the line generator 
 
  
  .curve(d3.curveMonotoneX);
 
  svg.append("path")
            .datum(data)
            .attr("d",line)
            .attr("stroke", color)
            .attr("stroke-width", Line_width)
            .attr("fill", "none");  
 
 
            svg.selectAll("dot")
            .data(data)
            .enter().append("circle")
            .attr("r", 1.5)

}
